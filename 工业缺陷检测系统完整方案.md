## 二、缺陷检测方案（MP157部署）

### 2.1 为什么YOLO不适合缺陷检测？

| 特性 | YOLO目标检测 | 缺陷检测需求 |
|------|-------------|-------------|
| 检测对象 | 有明确边界的物体 | 细微的表面异常 |
| 训练数据 | 需要大量标注 | 缺陷样本稀缺 |
| 检测粒度 | 物体级别 | 像素级别 |
| 适用场景 | 物体分类/定位 | 异常区域分割 |

### 2.2 推荐方案：传统机器视觉 + 轻量级深度学习混合

#### 方案A：传统机器视觉方法（适合MP157）

**核心算法组合：**

```
缺陷检测流程:
├── 预处理
│   ├── 高斯滤波去噪
│   ├── 直方图均衡化
│   └── 形态学操作
├── 特征提取
│   ├── 边缘检测 (Canny/Sobel)
│   ├── 纹理分析 (LBP/GLCM)
│   └── 颜色特征 (HSV/LAB)
└── 缺陷分割
    ├── 阈值分割 (OTSU/自适应)
    ├── 连通域分析
    └── 形态学处理
```

**OpenCV实现示例：**

```c
// 基于OpenCV的表面缺陷检测核心代码
#include <opencv2/opencv.hpp>
#include <opencv2/imgproc.hpp>

using namespace cv;

/**
 * @brief 表面缺陷检测主函数
 * @param src 输入图像
 * @param defects 输出缺陷掩码
 * @param defect_info 缺陷信息结构体
 */
typedef struct {
    int defect_count;      // 缺陷数量
    double defect_area;    // 缺陷总面积
    Rect defect_rect;      // 缺陷边界框
    int defect_type;       // 缺陷类型: 0-划痕, 1-凹坑, 2-锈蚀, 3-裂纹
} DefectInfo;

int surface_defect_detect(Mat src, Mat &defects, DefectInfo &info)
{
    Mat gray, blur, binary, morph;
    
    // Step 1: 灰度化
    if (src.channels() == 3) 
    {
        cvtColor(src, gray, COLOR_BGR2GRAY);
    }
    else 
    {
        gray = src.clone();
    }
    
    // Step 2: 高斯滤波去噪
    GaussianBlur(gray, blur, Size(5, 5), 0);
    
    // Step 3: 自适应阈值分割
    adaptiveThreshold(blur, binary, 255, 
                      ADAPTIVE_THRESH_GAUSSIAN_C,
                      THRESH_BINARY_INV, 11, 2);
    
    // Step 4: 形态学处理
    Mat kernel = getStructuringElement(MORPH_RECT, Size(3, 3));
    morphologyEx(binary, morph, MORPH_CLOSE, kernel);
    morphologyEx(morph, morph, MORPH_OPEN, kernel);
    
    // Step 5: 连通域分析
    Mat labels, stats, centroids;
    int num_labels = connectedComponentsWithStats(morph, labels, stats, centroids);
    
    // Step 6: 过滤小区域，提取缺陷
    defects = Mat::zeros(src.size(), CV_8UC1);
    info.defect_count = 0;
    info.defect_area = 0;
    
    for (int i = 1; i < num_labels; i++)
    {
        int area = stats.at<int>(i, CC_STAT_AREA);
        if (area > 50 && area < src.rows * src.cols * 0.3)  // 过滤噪点和过大区域
        {
            Mat mask = (labels == i);
            defects.setTo(255, mask);
            info.defect_count++;
            info.defect_area += area;
        }
    }
    
    return info.defect_count > 0 ? 1 : 0;
}

/**
 * @brief 划痕检测 - 基于边缘检测和霍夫变换
 */
int scratch_detect(Mat src, vector<Vec4i> &lines)
{
    Mat gray, edges;
    
    cvtColor(src, gray, COLOR_BGR2GRAY);
    GaussianBlur(gray, gray, Size(3, 3), 0);
    
    // Canny边缘检测
    Canny(gray, edges, 50, 150);
    
    // 霍夫直线检测
    HoughLinesP(edges, lines, 1, CV_PI/180, 50, 30, 10);
    
    return lines.size() > 0 ? 1 : 0;
}

/**
 * @brief 凹坑检测 - 基于形态学顶帽变换
 */
int pit_detect(Mat src, Mat &result)
{
    Mat gray, tophat, binary;
    
    cvtColor(src, gray, COLOR_BGR2GRAY);
    
    // 顶帽变换 - 提取暗点
    Mat kernel = getStructuringElement(MORPH_RECT, Size(15, 15));
    morphologyEx(gray, tophat, MORPH_TOPHAT, kernel);
    
    // 二值化
    threshold(tophat, binary, 30, 255, THRESH_BINARY);
    
    result = binary;
    return countNonZero(binary) > 100 ? 1 : 0;
}

/**
 * @brief 纹理异常检测 - 基于LBP局部二值模式
 */
Mat lbp_texture_detect(Mat src)
{
    Mat gray, lbp;
    
    cvtColor(src, gray, COLOR_BGR2GRAY);
    lbp = Mat::zeros(gray.size(), CV_8UC1);
    
    // LBP特征计算
    for (int i = 1; i < gray.rows-1; i++)
    {
        for (int j = 1; j < gray.cols-1; j++)
        {
            uchar center = gray.at<uchar>(i, j);
            uchar code = 0;
            
            code |= (gray.at<uchar>(i-1, j-1) > center) << 7;
            code |= (gray.at<uchar>(i-1, j) > center) << 6;
            code |= (gray.at<uchar>(i-1, j+1) > center) << 5;
            code |= (gray.at<uchar>(i, j+1) > center) << 4;
            code |= (gray.at<uchar>(i+1, j+1) > center) << 3;
            code |= (gray.at<uchar>(i+1, j) > center) << 2;
            code |= (gray.at<uchar>(i+1, j-1) > center) << 1;
            code |= (gray.at<uchar>(i, j-1) > center) << 0;
            
            lbp.at<uchar>(i, j) = code;
        }
    }
    
    return lbp;
}
```

**UNet-lite 部署示例（PyTorch → ONNX → NCNN）：**

```python
# train_unet_lite.py - 轻量级UNet训练
import torch
import torch.nn as nn

class DoubleConv(nn.Module):
    """双卷积块"""
    def __init__(self, in_ch, out_ch):
        super().__init__()
        self.conv = nn.Sequential(
            nn.Conv2d(in_ch, out_ch, 3, padding=1),
            nn.BatchNorm2d(out_ch),
            nn.ReLU(inplace=True),
            nn.Conv2d(out_ch, out_ch, 3, padding=1),
            nn.BatchNorm2d(out_ch),
            nn.ReLU(inplace=True)
        )
    
    def forward(self, x):
        return self.conv(x)

class UNetLite(nn.Module):
    """轻量级UNet - 针对嵌入式优化"""
    def __init__(self, in_ch=3, out_ch=1):
        super().__init__()
        
        # 编码器 - 使用深度可分离卷积减小参数量
        self.enc1 = DoubleConv(in_ch, 32)
        self.enc2 = DoubleConv(32, 64)
        self.enc3 = DoubleConv(64, 128)
        
        self.pool = nn.MaxPool2d(2)
        
        # 解码器
        self.dec3 = DoubleConv(128 + 64, 64)
        self.dec2 = DoubleConv(64 + 32, 32)
        self.dec1 = DoubleConv(32, 16)
        
        self.out_conv = nn.Conv2d(16, out_ch, 1)
        self.up = nn.Upsample(scale_factor=2, mode='bilinear', align_corners=True)
    
    def forward(self, x):
        # 编码
        e1 = self.enc1(x)
        e2 = self.enc2(self.pool(e1))
        e3 = self.enc3(self.pool(e2))
        
        # 解码
        d3 = self.dec3(torch.cat([self.up(e3), e2], dim=1))
        d2 = self.dec2(torch.cat([self.up(d3), e1], dim=1))
        d1 = self.dec1(self.up(d2))
        
        return torch.sigmoid(self.out_conv(d1))

# 导出ONNX模型
def export_to_onnx(model, input_size=(256, 256)):
    model.eval()
    dummy_input = torch.randn(1, 3, *input_size)
    
    torch.onnx.export(
        model,
        dummy_input,
        "unet_lite_defect.onnx",
        opset_version=11,
        input_names=['input'],
        output_names=['output'],
        dynamic_axes={'input': {0: 'batch'}, 'output': {0: 'batch'}}
    )
    print("ONNX模型导出成功！")

if __name__ == "__main__":
    model = UNetLite()
    export_to_onnx(model)
```

```c
// MP157 涡流检测数据接收处理
typedef struct {
    float amplitude;     // 幅值
    float phase;         // 相位
    float frequency;     // 频率
    int defect_flag;     // 缺陷标志
    char defect_type[32]; // 缺陷类型
} EddyCurrentData;

int parse_eddy_current(uint8_t *data, int len, EddyCurrentData *result)
{
    if (len < 16) return -1;
    
    // 解析涡流检测仪串口数据
    memcpy(&result->amplitude, data, 4);
    memcpy(&result->phase, data + 4, 4);
    memcpy(&result->frequency, data + 8, 4);
    memcpy(&result->defect_flag, data + 12, 4);
    
    // 根据相位变化判断缺陷类型
    if (result->phase > 30.0) {
        strcpy(result->defect_type, "裂纹");
    } else if (result->amplitude < 0.5) {
        strcpy(result->defect_type, "凹坑");
    } else {
        strcpy(result->defect_type, "正常");
    }
    
    return 0;
}
```

```c
/**
 * @brief 摄像头升降控制
 * @param target_mm 目标高度(mm)，范围0-100
 */
void camera_lift_control(float target_mm)
{
    float current_mm = 0;
    int steps = 0;
    
    // 计算步数：丝杆导程4mm，步进电机200步/圈，细分16
    // 每mm步数 = 200 * 16 / 4 = 800步/mm
    steps = (int)(target_mm * 800);
    
    // 设置方向
    if (target_mm > current_mm) 
    {
        HAL_GPIO_WritePin(DIR_GPIO_Port, DIR_Pin, GPIO_PIN_SET);  // 上升
    }
    else 
    {
        HAL_GPIO_WritePin(DIR_GPIO_Port, DIR_Pin, GPIO_PIN_RESET); // 下降
        steps = -steps;
    }
    
    // 执行步进
    for (int i = 0; i < abs(steps); i++)
    {
        HAL_GPIO_WritePin(PUL_GPIO_Port, PUL_Pin, GPIO_PIN_SET);
        HAL_Delay_us(500);  // 脉冲宽度
        HAL_GPIO_WritePin(PUL_GPIO_Port, PUL_Pin, GPIO_PIN_RESET);
        HAL_Delay_us(500);
    }
    
    current_mm = target_mm;
}

/**
 * @brief MP157发送摄像头高度指令
 */
void mp157_send_camera_height(uint8_t height_level)
{
    uint8_t cmd[5] = {0xAA, 0x05, 0x01, height_level, 0x55};
    HAL_UART_Transmit(&huart1, cmd, 5, 100);
}
```

### 6.4 检测记录时间字段设计

为了避免“检测发生时间”和“云端上传完成时间”混淆，检测记录不要只保存一个 `detect_time`，建议统一为以下字段：

| 字段 | 含义 | 是否必须 | 说明 |
|------|------|---------|------|
| `captured_at` | 相机拍到用于检测的原始图像时间 | 必须 | 这是最重要的业务时间，表示检测实际发生的时间 |
| `detected_at` | 模型完成判定、生成结果的时间 | 建议 | 如果检测流程很快，可与 `captured_at` 相同或非常接近 |
| `uploaded_at` | 图片或结果图真正上传完成并持久化到云端的时间 | 必须 | 代表“云端接收完成时间”，不等于检测实际发生时间 |
| `storage_last_modified` | COS 对象最后修改时间 | 可选 | 可作为 COS 侧校验值，不建议单独用它做业务排序 |

推荐排序规则：

| 使用场景 | 推荐字段 |
|------|---------|
| 历史记录默认排序 | `captured_at DESC` |
| 查看云端同步延迟 | `uploaded_at` 与 `captured_at` 的差值 |
| 对账 COS 时间 | `storage_last_modified` |

说明：

1. 如果 MP157 检测完成后立刻上传，那么 `captured_at` 与 `detected_at` 差距通常很小。
2. 但 `uploaded_at` 仍然需要独立保存，因为网络延迟、断点补传、服务器重试都会让它晚于检测时间。
3. COS 自身会记录对象最后修改时间，但该时间只能说明“对象何时落到存储”，不能替代业务上的检测时间。

```
   - 涡流检测模块探测内部裂纹与凹坑

2. 运动控制层：
   - STM32F4作为下位机控制传送带与机械臂
   - 通过串口接收MP157坐标指令
   - 实现零件精确定位与分拣

3. 人机交互层：
   - 基于LVGL开发7寸触摸屏界面
   - 实时显示检测画面、结果标注、统计数据

4. 云端服务层：
   - 部署FastAPI后端与Vue前端
   - 存储检测历史数据与图片
   - 调用大模型API进行二次分析
```

### 7.4 创新点

```
1. 双层检测架构：边缘端实时初检 + 云端AI确认，兼顾效率与准确性

2. 多模态缺陷检测：融合可见光图像与涡流信号，实现表里兼顾的全面检测

3. 轻量化部署：针对MP157算力优化检测算法，无需GPU即可实时运行

4. 完整自动化流程：实现上料→检测→分拣全流程无人值守
```

## 八、服务器部署详细步骤

### 8.1 基础环境安装

```bash
# 1. 更新系统
apt update && apt upgrade -y

# 2. 安装基础工具
apt install -y git curl wget vim nginx

# 3. 安装Python环境
apt install -y python3 python3-pip python3-venv

# 4. 安装Node.js (前端构建)
curl -fsSL https://deb.nodesource.com/setup_18.x | bash -
apt install -y nodejs

# 5. 安装MySQL
apt install -y mysql-server
mysql_secure_installation

# 6. 安装Redis
apt install -y redis-server

# 7. 配置腾讯云 COS
pip install cos-python-sdk-v5
# 在 .env 中配置 COS_SECRET_ID、COS_SECRET_KEY、COS_REGION、COS_BUCKET
```

### 8.2 后端部署

```bash
# 1. 克隆代码
cd /opt
git clone https://github.com/your-repo/defect-detection.git
cd defect-detection/backend

# 2. 创建虚拟环境
python3 -m venv venv
source venv/bin/activate

# 3. 安装依赖
pip install -r requirements.txt

# 4. 配置环境变量
cp .env.example .env
vim .env
# 编辑数据库连接、API密钥等

# 5. 初始化数据库
python init_db.py

# 6. 使用Gunicorn启动（生产环境）
pip install gunicorn
gunicorn -w 4 -k uvicorn.workers.UvicornWorker \
    main:app --bind 0.0.0.0:8000 --daemon
```

### 8.3 前端部署

```bash
# 1. 进入前端目录
cd ../frontend

# 2. 安装依赖
npm install

# 3. 配置API地址
vim .env.production
# VITE_API_BASE_URL=https://your-domain.com/api

# 4. 构建
npm run build

# 5. 部署到Nginx
cp -r dist/* /var/www/html/
```

### 8.4 Nginx配置

```nginx
# /etc/nginx/sites-available/defect-detection
server {
    listen 80;
    server_name your-domain.com;
    
    # 前端静态文件
    location / {
        root /var/www/html;
        try_files $uri $uri/ /index.html;
    }
    
    # 后端API代理
    location /api {
        proxy_pass http://127.0.0.1:8000;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
    }
    
    # 图片文件访问
    location /uploads {
        alias /opt/defect-detection/backend/uploads;
    }
}
```

### 8.5 HTTPS配置（可选）

```bash
# 安装Certbot
apt install -y certbot python3-certbot-nginx

# 申请证书
certbot --nginx -d your-domain.com

# 自动续期
certbot renew --dry-run
```
